Name of MVP matrix in shader must be MVP_matrix
    uniform mat4 MVP_matrix;

Name of model matrix in shader must be M_matrix
    uniform mat4 M_matrix;

Fragment shader can contain 5 sampler2D for textures.
    uniform sampler2D diffuseTexture;   // name and order are important
    uniform sampler2D specularTexture;  // name and order are important
    uniform sampler2D shadowMapTexture; // name and order are important
    uniform sampler2D normalMapTexture; // name and order are important
    uniform sampler2D heightMapTexture; // name and order are important

Vertex shader example:
    #version 300 es

    precision highp float; // highp mediump lowp

    layout(location = 0) in vec3 inPosition;
    layout(location = 1) in vec3 inNormal;
    layout(location = 2) in vec2 inTextureCoords;

    out vec2 textureCoords;

    uniform mat4 MVP_matrix;

    void main()
    {
        textureCoords = inTextureCoords;

        gl_Position = MVP_matrix * vec4(inPosition, 1.0);
    }

Fragment shader example:
    #version 300 es

    precision highp float; // highp mediump lowp

    in vec2 textureCoords;

    out vec4 outColor;

    uniform sampler2D diffuseTexture;   // first in shader !!!
    uniform sampler2D specularTexture;  // second in shader !!!

    void main()
    {
        outColor = texture(diffuseTexture, textureCoords);
    }
